```python
# Simplified example of supervised learning for email classification
# In reality, this would use libraries like scikit-learn

# Training data: emails with labels
training_emails = [
    {"text": "Win free money now!", "label": "spam"},
    {"text": "Meeting at 3pm tomorrow", "label": "legitimate"},
    {"text": "Click here for prizes", "label": "spam"},
    {"text": "Project deadline reminder", "label": "legitimate"}
]

# Extract features from emails (simplified)
def extract_features(email_text):
    """Extract simple features from email text"""
    features = {
        "has_free": "free" in email_text.lower(),
        "has_click": "click" in email_text.lower(),
        "has_win": "win" in email_text.lower(),
        "length": len(email_text)
    }
    return features

# Train the model (conceptual - showing the process)
def train_model(training_data):
    """Learn patterns from labelled training data"""
    spam_features = []
    legitimate_features = []
    
    for email in training_data:
        features = extract_features(email["text"])
        if email["label"] == "spam":
            spam_features.append(features)
        else:
            legitimate_features.append(features)
    
    return {"spam_patterns": spam_features, "legitimate_patterns": legitimate_features}

# Use the trained model to classify new emails
def classify_email(email_text, model):
    """Predict whether a new email is spam or legitimate"""
    features = extract_features(email_text)
    
    # Simple classification logic (real models are more sophisticated)
    spam_score = 0
    if features["has_free"] or features["has_click"] or features["has_win"]:
        spam_score += 1
    
    return "spam" if spam_score > 0 else "legitimate"

# Train the model
model = train_model(training_emails)

# Test on new email
new_email = "Free prize for you!"
prediction = classify_email(new_email, model)
print(f"Email classified as: {prediction}")  # Output: spam
```

```python
def calculate_distance(customer1, customer2):
    """Calculate similarity between two customers"""
    # Simple Euclidean distance (simplified)
    age_diff = (customer1["age"] - customer2["age"]) ** 2
    purchase_diff = (customer1["purchases_per_month"] - customer2["purchases_per_month"]) ** 2
    spend_diff = (customer1["avg_spend"] - customer2["avg_spend"]) ** 2
    
    return (age_diff + purchase_diff + spend_diff) ** 0.5

def simple_clustering(customers, num_clusters=2):
    """Group customers into clusters based on similarity"""
    # Simplified K-means concept
    clusters = [[] for _ in range(num_clusters)]
    
    # Assign each customer to nearest cluster (simplified)
    for customer in customers:
        # In real K-means, we'd calculate distance to cluster centres
        # Here we use a simple heuristic based on spending
        if customer["avg_spend"] > 100:
            clusters[0].append(customer)  # High spenders
        else:
            clusters[1].append(customer)  # Regular spenders
    
    return clusters
```

## Example: Simple Game Agent
```python
# Simplified reinforcement learning for a grid navigation game
import random

class GridWorld:
    """Simple 5x5 grid where agent learns to reach a goal"""
    def __init__(self):
        self.grid_size = 5
        self.agent_pos = [0, 0]  # Start position
        self.goal_pos = [4, 4]   # Goal position
        
    def reset(self):
        """Reset agent to starting position"""
        self.agent_pos = [0, 0]
        return self.agent_pos.copy()
    
    def step(self, action):
        """Take an action and return new state and reward"""
        # Actions: 0=up, 1=right, 2=down, 3=left
        moves = [[-1, 0], [0, 1], [1, 0], [0, -1]]
        
        # Try to move
        new_pos = [
            self.agent_pos[0] + moves[action][0],
            self.agent_pos[1] + moves[action][1]
        ]
        
        # Check if move is valid (within grid)
        if 0 <= new_pos[0] < self.grid_size and 0 <= new_pos[1] < self.grid_size:
            self.agent_pos = new_pos
        
        # Calculate reward
        if self.agent_pos == self.goal_pos:
            reward = 100  # Large reward for reaching goal
            done = True
        else:
            reward = -1   # Small penalty for each step (encourages efficiency)
            done = False
        
        return self.agent_pos.copy(), reward, done

class SimpleAgent:
    """Agent that learns through reinforcement"""
    def __init__(self):
        # Q-table stores expected rewards for each state-action pair
        self.q_table = {}
        self.learning_rate = 0.1
        self.discount_factor = 0.9
        self.exploration_rate = 0.3
    
    def get_q_value(self, state, action):
        """Get expected reward for state-action pair"""
        state_key = tuple(state)
        if state_key not in self.q_table:
            self.q_table[state_key] = [0, 0, 0, 0]  # 4 possible actions
        return self.q_table[state_key][action]
    
    def choose_action(self, state):
        """Choose action using exploration vs exploitation"""
        # Exploration: try random actions
        if random.random() < self.exploration_rate:
            return random.randint(0, 3)
        
        # Exploitation: choose best known action
        state_key = tuple(state)
        if state_key not in self.q_table:
            return random.randint(0, 3)
        
        return self.q_table[state_key].index(max(self.q_table[state_key]))
    
    def learn(self, state, action, reward, next_state):
        """Update knowledge based on experience"""
        state_key = tuple(state)
        if state_key not in self.q_table:
            self.q_table[state_key] = [0, 0, 0, 0]
        
        # Q-learning update rule
        current_q = self.q_table[state_key][action]
        max_next_q = max([self.get_q_value(next_state, a) for a in range(4)])
        new_q = current_q + self.learning_rate * (reward + self.discount_factor * max_next_q - current_q)
        self.q_table[state_key][action] = new_q

# Train the agent
env = GridWorld()
agent = SimpleAgent()

print("Training agent...")
for episode in range(100):
    state = env.reset()
    total_reward = 0
    
    for step in range(50):  # Maximum 50 steps per episode
        action = agent.choose_action(state)
        next_state, reward, done = env.step(action)
        agent.learn(state, action, reward, next_state)
        
        total_reward += reward
        state = next_state
        
        if done:
            break
    
    if (episode + 1) % 20 == 0:
        print(f"Episode {episode + 1}: Total reward = {total_reward}")

print("\nAgent trained! It learned to navigate to the goal.")
```

```python
# Conceptual example of transfer learning
# Real implementations use frameworks like TensorFlow or PyTorch

class PreTrainedModel:
    """Represents a model trained on millions of general images"""
    def __init__(self):
        self.name = "ImageNet Pre-trained Model"
        self.trained_on = "1.4 million images, 1000 categories"
        self.layers = [
            "Convolutional layers (feature extraction)",
            "Pooling layers",
            "Fully connected layers",
            "Output layer (1000 classes)"
        ]
    
    def extract_features(self, image):
        """Use pre-trained layers to extract features"""
        print(f"Extracting features using knowledge from {self.trained_on}")
        return "High-level image features"

class TransferLearningModel:
    """Adapt pre-trained model for new task"""
    def __init__(self, base_model, new_task):
        self.base_model = base_model
        self.new_task = new_task
        self.custom_layers = []
    
    def adapt_for_new_task(self, num_classes):
        """Modify model for specific task"""
        print(f"\n--- Adapting model for: {self.new_task} ---")
        
        # Keep pre-trained feature extraction layers
        print("✓ Keeping pre-trained convolutional layers (frozen)")
        print("  These layers already know how to detect edges, shapes, textures")
        
        # Remove original output layer
        print("✗ Removing original output layer (1000 classes)")
        
        # Add new custom layers
        print(f"+ Adding new output layer ({num_classes} classes)")
        self.custom_layers.append(f"Custom output layer: {num_classes} classes")
        
        print(f"\nModel ready for {self.new_task}!")
    
    def train(self, training_data_size):
        """Train the adapted model"""
        print(f"\nTraining with {training_data_size} images")
        print("Only training new layers (much faster!)")
        print("Pre-trained layers provide general image understanding")
        return "Training complete"
    
    def compare_approaches(self):
        """Show benefits of transfer learning"""
        print("\n--- Comparison ---")
        print("Training from scratch:")
        print("  • Needs: 100,000+ images")
        print("  • Time: Days or weeks")
        print("  • Risk: Overfitting with small datasets")
        
        print("\nTransfer learning:")
        print("  • Needs: 1,000-10,000 images")
        print("  • Time: Hours")
        print("  • Benefit: Leverages existing knowledge")

# Demonstrate transfer learning
print("=== Transfer Learning Example ===\n")

# Start with pre-trained model
base_model = PreTrainedModel()
print(f"Base model: {base_model.name}")
print(f"Trained on: {base_model.trained_on}\n")

# Adapt for medical imaging task
medical_model = TransferLearningModel(base_model, "Detecting pneumonia in chest X-rays")
medical_model.adapt_for_new_task(num_classes=2)  # Healthy vs Pneumonia
medical_model.train(training_data_size=5000)
medical_model.compare_approaches()

# Another example: custom object detection
```
